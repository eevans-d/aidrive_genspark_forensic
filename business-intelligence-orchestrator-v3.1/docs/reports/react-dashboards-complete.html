import React, { useEffect, useRef } from 'react'; import * as d3 from 'd3'; interface D3ChartsProps { type: 'line' | 'bar' | 'donut' | 'heatmap' | 'network' | 'combinedChart'; data: any[]; config?: { width?: number; const containerRef = useRef<HTMLDivElement>(null); const defaultConfig = { width: 800, height: 400, margin: { top: 20, right: 30, bottom: 40, left: 50 }, showTooltip: true, responsive: true, animations: true, ...config }; useEffect(() => { if (!data || !svgRef.current) return; const svg = d3.select(svgRef.current); svg.selectAll("*").remove(); switch (type) { case 'line': renderLineChart(svg, data, defaultConfig); break; case 'bar': renderBarChart(svg, data, defaultConfig); break; case 'donut': renderDonutChart(svg, data, defaultConfig); break; case 'combinedChart': renderCombinedChart(svg, data, defaultConfig); break; // Animation if (config.animations) { const totalLength = path.node()?.getTotalLength() || 0; path .attr('stroke-dasharray', `${totalLength} ${totalLength}`) .attr('stroke-dashoffset', totalLength) .transition() .duration(2000) .attr('stroke-dashoffset', 0); } // Tooltip if (config.showTooltip) { const tooltip = d3.select('body').append('div') .attr('class', 'tooltip') .style('opacity', 0) .style('position', 'absolute') .style('background', 'rgba(0, 0, 0, 0.8)') .style('color', 'white') .style('padding', '8px') .style('border-radius', '4px') .style('pointer-events', 'none'); }); } }; const renderDonutChart = (svg: any, data: any[], config: any) => { const { width, height, innerRadius = 60, outerRadius = 120 } = config; const radius = Math.min(width, height) / 2; const color = d3.scaleOrdinal() .domain(data.map(d => d.label)) .range(d3.schemeSet3); const arcs = g.selectAll('.arc') .data(pie(data)) .enter().append('g') .attr('class', 'arc'); arcs.append('path') .attr('d', arc) .attr('fill', d => color(d.data.label)) .attr('stroke', 'white') .attr('stroke-width', 2) .transition() .duration(1000) .attrTween('d', function(d) { const interpolate = d3.interpolate({startAngle: 0, endAngle: 0}, d); } }; const renderCombinedChart = (svg: any, data: any[], config: any) => { // Implementation for combined line + bar chart // This would include multiple datasets and scales }; return ( <div ref={containerRef} className="w-full"> <svg ref={svgRef} width={config.responsive ? '100%' : defaultConfig.width} height={defaultConfig.height} viewBox={config.responsive ? `0 0 ${defaultConfig.width} ${defaultConfig.height}` : undefined} className="d3-chart" /> </div> ); }; export default D3Charts;
