# ================================================================
# BUSINESS INTELLIGENCE ORCHESTRATOR v3.1
# Docker Compose - Orquestación Completa de Servicios
# ================================================================

version: '3.8'

services:

  # ================================
  # BASE DE DATOS PRINCIPAL
  # ================================
  postgres:
    image: postgres:15-alpine
    container_name: bi_postgres
    restart: unless-stopped
    environment:
      POSTGRES_DB: bi_orchestrator
      POSTGRES_USER: bi_admin
      POSTGRES_PASSWORD: BI_Secure_2024!
      POSTGRES_INITDB_ARGS: --encoding=UTF-8 --lc-collate=C --lc-ctype=C
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./supabase/schema.sql:/docker-entrypoint-initdb.d/01-schema.sql
      - ./supabase/functions.sql:/docker-entrypoint-initdb.d/02-functions.sql
    ports:
      - "5432:5432"
    networks:
      - bi_backend
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U bi_admin -d bi_orchestrator"]
      interval: 30s
      timeout: 10s
      retries: 5

  # ================================
  # REDIS CLUSTER PARA CACHING
  # ================================
  redis-master:
    image: redis:7-alpine
    container_name: bi_redis_master
    restart: unless-stopped
    command: redis-server --appendonly yes --maxmemory 2gb --maxmemory-policy allkeys-lru
    volumes:
      - redis_master_data:/data
    ports:
      - "6379:6379"
    networks:
      - bi_backend

  redis-slave:
    image: redis:7-alpine
    container_name: bi_redis_slave
    restart: unless-stopped
    command: redis-server --slaveof redis-master 6379 --appendonly yes
    volumes:
      - redis_slave_data:/data
    depends_on:
      - redis-master
    networks:
      - bi_backend

  # ================================
  # N8N WORKFLOW AUTOMATION
  # ================================
  n8n:
    image: n8nio/n8n:latest
    container_name: bi_n8n
    restart: unless-stopped
    environment:
      DB_TYPE: postgresdb
      DB_POSTGRESDB_HOST: postgres
      DB_POSTGRESDB_PORT: 5432
      DB_POSTGRESDB_DATABASE: n8n
      DB_POSTGRESDB_USER: bi_admin
      DB_POSTGRESDB_PASSWORD: BI_Secure_2024!
      N8N_BASIC_AUTH_ACTIVE: true
      N8N_BASIC_AUTH_USER: admin
      N8N_BASIC_AUTH_PASSWORD: BI_N8N_2024!
      WEBHOOK_URL: http://localhost:5678
      N8N_METRICS: true
    volumes:
      - n8n_data:/home/node/.n8n
      - ./n8n-workflows:/home/node/.n8n/workflows
    ports:
      - "5678:5678"
    depends_on:
      - postgres
      - redis-master
    networks:
      - bi_backend
      - bi_frontend

  # ================================
  # WEB AUTOMÁTICO COMPETITIVO TIER 2.5
  # ================================
  web-competitive-monitor:
    build:
      context: .
      dockerfile: Dockerfile.web-monitor
    container_name: bi_web_monitor
    restart: unless-stopped
    environment:
      POSTGRES_HOST: postgres
      POSTGRES_DB: bi_orchestrator
      POSTGRES_USER: bi_admin
      POSTGRES_PASSWORD: BI_Secure_2024!
      REDIS_HOST: redis-master
      REDIS_PORT: 6379
      PROXY_ROTATION: true
      MONITORING_INTERVAL: 900
      ALERT_WEBHOOKS: http://n8n:5678/webhook/competitive-alert
    volumes:
      - ./proxy-management:/app/proxies
      - competitive_data:/app/data
    depends_on:
      - postgres
      - redis-master
    networks:
      - bi_backend
      - bi_external

  # ================================
  # MICROSERVICIOS BACKEND
  # ================================
  api-gateway:
    build:
      context: ./backend/api-gateway
    container_name: bi_api_gateway
    restart: unless-stopped
    environment:
      REDIS_URL: redis://redis-master:6379
      JWT_SECRET: BI_JWT_Secret_Key_2024_Ultra_Secure!
      RATE_LIMIT_REQUESTS: 1000
      RATE_LIMIT_WINDOW: 3600
    ports:
      - "8000:8000"
    depends_on:
      - redis-master
    networks:
      - bi_backend
      - bi_frontend

  data-analytics:
    build:
      context: ./backend/data-analytics
    container_name: bi_data_analytics
    restart: unless-stopped
    environment:
      POSTGRES_URL: postgresql://bi_admin:BI_Secure_2024!@postgres:5432/bi_orchestrator
      REDIS_URL: redis://redis-master:6379
      ML_MODEL_PATH: /app/models
    volumes:
      - ml_models:/app/models
    depends_on:
      - postgres
      - redis-master
    networks:
      - bi_backend

  competitive-intelligence:
    build:
      context: ./backend/competitive-intelligence
    container_name: bi_competitive_intel
    restart: unless-stopped
    environment:
      POSTGRES_URL: postgresql://bi_admin:BI_Secure_2024!@postgres:5432/bi_orchestrator
      REDIS_URL: redis://redis-master:6379
      SENTIMENT_API_KEY: your_sentiment_api_key
      WEB_MONITOR_URL: http://web-competitive-monitor:8080
    depends_on:
      - postgres
      - web-competitive-monitor
    networks:
      - bi_backend

  business-metrics:
    build:
      context: ./backend/business-metrics
    container_name: bi_business_metrics
    restart: unless-stopped
    environment:
      POSTGRES_URL: postgresql://bi_admin:BI_Secure_2024!@postgres:5432/bi_orchestrator
      REDIS_URL: redis://redis-master:6379
      REPORTS_SCHEDULE: 0 8 * * *
    depends_on:
      - postgres
      - redis-master
    networks:
      - bi_backend

  # ================================
  # MONITOREO Y VISUALIZACIÓN
  # ================================
  grafana:
    image: grafana/grafana:latest
    container_name: bi_grafana
    restart: unless-stopped
    environment:
      GF_SECURITY_ADMIN_PASSWORD: BI_Grafana_2024!
      GF_DATABASE_TYPE: postgres
      GF_DATABASE_HOST: postgres:5432
      GF_DATABASE_NAME: grafana
      GF_DATABASE_USER: bi_admin
      GF_DATABASE_PASSWORD: BI_Secure_2024!
    volumes:
      - grafana_data:/var/lib/grafana
      - ./monitoring/grafana/dashboards:/etc/grafana/provisioning/dashboards
      - ./monitoring/grafana/datasources:/etc/grafana/provisioning/datasources
    ports:
      - "3000:3000"
    depends_on:
      - postgres
    networks:
      - bi_backend
      - bi_frontend

  loki:
    image: grafana/loki:latest
    container_name: bi_loki
    restart: unless-stopped
    command: -config.file=/etc/loki/local-config.yaml
    volumes:
      - loki_data:/loki
    ports:
      - "3100:3100"
    networks:
      - bi_backend

  # ================================
  # FRONTEND APLICACIÓN
  # ================================
  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile.prod
    container_name: bi_frontend
    restart: unless-stopped
    environment:
      REACT_APP_API_URL: http://localhost:8000
      REACT_APP_WS_URL: ws://localhost:8001
      REACT_APP_N8N_URL: http://localhost:5678
    ports:
      - "3001:80"
    depends_on:
      - api-gateway
    networks:
      - bi_frontend

# ================================
# CONFIGURACIÓN DE REDES
# ================================
networks:
  bi_backend:
    driver: bridge
    internal: true
  bi_frontend:
    driver: bridge
  bi_external:
    driver: bridge

# ================================
# VOLÚMENES PERSISTENTES
# ================================
volumes:
  postgres_data:
    driver: local
  redis_master_data:
    driver: local
  redis_slave_data:
    driver: local
  n8n_data:
    driver: local
  competitive_data:
    driver: local
  ml_models:
    driver: local
  grafana_data:
    driver: local
  loki_data:
    driver: local
